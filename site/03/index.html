    控制流

    if 语句

    基本语法

    import t

def main() -> t.CInt:
    x: t.CInt = 10

    if x > 5:
        x = x + 1

    return 0

    编译后：

    int main(void) {
    int x = 10;

    if ((x > 5)) {
        x = (x + 1);
    }

    return 0;
}

    if-elif-else

    import t

def main() -> t.CInt:
    x: t.CInt = 10

    if x > 10:
        x = 20
    elif x > 5:
        x = 15
    elif x > 0:
        x = 5
    else:
        x = 0

    return 0

    编译后：

    int main(void) {
    int x = 10;

    if ((x > 10)) {
        x = 20;
    } else if ((x > 5)) {
        x = 15;
    } else if ((x > 0)) {
        x = 5;
    } else {
        x = 0;
    }

    return 0;
}

    嵌套 if

    import t

def main() -> t.CInt:

    y: t.CInt = 20

            x = x + y
        else:
            x = x - y

    return 0

    编译后：

    int main(void) {
    int x = 10;
    int y = 20;

    if ((x > 5)) {
        if ((y > 10)) {
            x = (x + y);
        } else {
            x = (x - y);
        }
    }

    return 0;
}

    do-while 循环

    TransPyC 使用特殊的语法表示 do-while 循环：

    import t, c

def main() -> t.CInt:
    i: t.CInt = 0

    c.DoWhile(i < 10)
        i = i + 1

    return 0

    编译后：

    int main(void) {
    int i = 0;

    do {
        i = (i + 1);
    } while ((i < 10));

    return 0;
}

    条件表达式

    比较运算符

    import t

def main() -> t.CInt:
    a: t.CInt = 10
    b: t.CInt = 20

    eq: t.CInt = a == b    # 等于
    ne: t.CInt = a != b    # 不等于
    lt: t.CInt = a < b     # 小于
    le: t.CInt = a <= b    # 小于等于
    gt: t.CInt = a > b     # 大于
    ge: t.CInt = a >= b    # 大于等于

    return 0

    编译后：

    int main(void) {
    int a = 10;
    int b = 20;

    int eq = (a == b);
    int ne = (a != b);
    int lt = (a < b);
    int le = (a <= b);
    int gt = (a > b);
    int ge = (a >= b);

    return 0;
}

    return 0

    int and_bits = (a & b);
    int or_bits = (a | b);
    int xor_bits = (a ^ b);
    int not_bits = (~a);
    int left_shift = (a << 2);
    int right_shift = (a >> 2);

    return 0;
}

    switch 语句

    TransPyC 使用特殊的语法表示 switch 语句：

    import t, c

def main() -> t.CInt:
    x: t.CInt = 2

    c.Switch(x)
    c.Case(1)
        x = 10
        c.Break()
    c.Case(2)
        x = 20
        c.Break()
    c.Case(3)
        x = 30
        c.Break()
    c.Default()
        x = 0
        c.Break()

    return 0

    编译后：

    int main(void) {
    int x = 2;

    switch (x) {
        case 1:
            x = 10;
            break;
        case 2:
            x = 20;
            break;
        case 3:
            x = 30;
            break;
        default:
            x = 0;
            break;
    }

    return 0;
}

    goto 语句

    TransPyC 支持 goto 语句：

    import t, c

def main() -> t.CInt:
    i: t.CInt = 0

    c.Label("loop")
    i = i + 1
    if i < 10:
        c.Goto("loop")

    return 0

    编译后：

    int main(void) {
    int i = 0;

loop:
    i = (i + 1);
    if ((i < 10)) {
        goto loop;
    }

    return 0;
}
